11// Fonction pour calculer le gain de direction en fonction de l'angle
12double gain_steer(double x) {
13 if (x > 100.0) { // Si x est supérieur à 100
14 x = 100.0; // On fixe x à 100
15 }
16 int q = (int)((x-fmod(x,10.0)) / 10); // On calcule la partie entière de x/10
17 return (steers[q + 1] - steers[q]) / 10.0 * (x - (double)q * 10.0) + steers[q]; // On retourne le gain de direction calculé
18}

106 double speed = 0.0; // Initialise speed à 0
107 DVecteur vecteurs; // Déclare une variable vecteurs de type DVecteur
108 pixyVecteursToDVecteurs(ctx->pixy_vector,&vecteurs); // Convertit les vecteurs Pixy en DVecteurs

110 int num_vectors = getNbOfNonNullVecteurs(&vecteurs); // Obtient le nombre de vecteurs non nuls
111 arrangeVecteurs(&vecteurs); // Arrange les vecteurs

113 if(ctx->nb_of_vects_stored == NB_OF_VECTS_TO_WAIT){ // Si le nombre de vecteurs stockés est égal à NB_OF_VECTS_TO_WAIT
114 DVecteur average_vecteurs; // Déclare une variable average_vecteurs de type DVecteur
115 setNullVecteur(&average_vecteurs); // Met à zéro average_vecteurs
116 calculateAverageVecteursWithoutOutliers(ctx->oldVectors,&average_vecteurs,NB_OF_VECTS_TO_WAIT); // Calcule la moyenne des vecteurs sans les valeurs aberrantes
117 removeSmallVecteurs(&average_vecteurs); // Supprime les petits vecteurs
118 removeLargeAngleVecteur(&average_vecteurs); // Supprime les vecteurs avec un grand angle
119 mergeSameVecteurs(&average_vecteurs); // Fusionne les vecteurs similaires dans average_vecteurs
120 num_vectors = getNbOfNonNullVecteurs(&average_vecteurs); // Obtient le nombre de vecteurs non nuls dans average_vecteurs
121
122 // Si aucun vecteur n'est détecté, on arrête la voiture
123 if(num_vectors == 0) {
124 // Si c'est la première fois qu'aucun vecteur n'est détecté
125 if(ctx->no_line_time == 0){
126 ctx->no_line_time=k_uptime_get(); // Enregistre le temps actuel
127 steer = ctx->prev_steer; // Utilise la dernière valeur de direction
128 speed=NO_VECTOR_SPEED; // Définit la vitesse à NO_VECTOR_SPEED
129 LOG_INF("\r\nFirst vector\r\n"); // Affiche un message d'information
130 }
131 // Si cela fait plus de NO_LINE_TIME ms qu'aucun vecteur n'a été détecté
132 else if(k_uptime_get() - ctx->no_line_time >= NO_LINE_TIME) {
133 speed = 0.0; // Arrête la voiture
134 steer = 0.0; // Arrête de tourner
135 LOG_INF("\r\n No vectors, stopping"); // Affiche un message d'information
136 }
137 else{
138 LOG_INF("\r\n %lld ms without any vectors", k_uptime_get() - ctx->no_line_time); // Affiche le temps écoulé depuis la dernière détection de vecteur
139 speed = NO_VECTOR_SPEED; // Continue à la même vitesse
140 steer =ctx->prev_steer; // Continue à tourner dans la même direction
141 }
142
143 // Si la direction est supérieure à INCREASE_STEER_IF_NO_VECTORS_THRESHOLD
144 if(steer > INCREASE_STEER_IF_NO_VECTORS_THRESHOLD){
145 steer+=INCREASE_STEER_IF_NO_VECTORS; // Augmente la direction
146 }
147 // Si la direction est inférieure à -INCREASE_STEER_IF_NO_VECTORS_THRESHOLD
148 else if(steer < -INCREASE_STEER_IF_NO_VECTORS_THRESHOLD){
149 steer-=INCREASE_STEER_IF_NO_VECTORS; // Diminue la direction
150 }
151 }
152 // Si deux vecteurs sont détectés et que le premier vecteur n'a pas encore été enregistré
153 else if(num_vectors == 2 && ctx->first_vector_saved == 0){
154 maximizeDVector(&average_vecteurs); // Maximise average_vecteurs
155 saveReferenceVecteur(ctx,average_vecteurs); // Enregistre le vecteur de référence
156 }
157 else{
158 // Si un seul vecteur est détecté et que le premier vecteur a été enregistré
159 if(num_vectors == 1 && ctx->first_vector_saved == 1){
160 buildSecondVecteur(&average_vecteurs,ctx); // Construit le second vecteur
161 num_vectors++; // Incrémente le nombre de vecteurs
162 }
163 // Si deux vecteurs sont détectés
164 if(num_vectors == 2){
165 ctx->no_line_time = 0; // Réinitialise le temps sans détection de ligne
166
167 double angle_antoine = (getMiddleAngle(average_vecteurs) -ctx->angle_ref)*100/M_PI; // Calcule l'angle du vecteur central moins l'erreur statique du début et met cet angle en %
168 double angle_percent = angle_antoine*100.0/15.0; // Applique un premier gain pour augmenter les petits angles
169
170 double steer_gain = gain_steer(fabs(angle_percent)):1.0; // Calcule le gain de direction
171 steer = steer_gain*angle_percent; // Calcule la nouvelle direction
172
173 LOG_INF("angle: %f, angle(percent): %f, gain: %f, steer : %f",angle_antoine,angle_percent,steer_gain,steer); // Affiche les informations de direction
174 }
175 steer= steer_saturation(steer); // Sature la direction
176 ctx->prev_steer = steer; // Enregistre la direction actuelle
177 speed = gain_speed(fabs(x)); // Calcule la vitesse
178 speed = speed_saturation(speed); // Sature la vitesse
179 steer = steer*STEER_FACTOR; // Applique le facteur de direction
180 ctx->prev_speed = speed; // Enregistre la vitesse actuelle
181 double omega_fwd=speed*SPEED_FACTOR; // Calcule la vitesse avant
182
183 b3rb_set_actuators(&ctx->actuators, steer, omega_fwd); // Met à jour les actionneurs
184 }
