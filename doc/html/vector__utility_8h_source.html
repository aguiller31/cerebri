501void buildSecondVecteur(DVecteur* vecteurs, context* ctx) {
502 Vecteur working_vecteur = vecteurs->m[0];
503
504 double l = 1.0;
505 double ref_angle = ctx->beta_ref;
506 Vecteur ref_vecteur = ctx->first_vector.m[0];
507
508 if (getVecteurSide(working_vecteur, IMAGE_WIDTH / 2) !=
509 VECTOR_ON_THE_LEFT) {
510 l = -1.0;
511 ref_angle = ctx->sigma_ref;
512 ref_vecteur = ctx->first_vector.m[1];
513 }
514
515 double omega =
516 getPointDistance(working_vecteur.head, working_vecteur.tail) /
517 getPointDistance(ref_vecteur.head, ref_vecteur.tail);
518 double dist_head = omega * ctx->head_distance_ref;
519 double beta = atan2(working_vecteur.head.x - working_vecteur.tail.x,
520 working_vecteur.head.y - working_vecteur.tail.y);
521 double angle1 = M_PI + l * beta - fabs(ref_angle);
522 vecteurs->m[1].head.y = working_vecteur.head.y + cos(angle1) * dist_head;
523 vecteurs->m[1].head.x =
524 working_vecteur.head.x + l * sin(angle1) * dist_head;
525
526 double dist_tail = omega * ctx->tail_distance_ref;
527 double angle2 = M_PI / 2 - angle1;
528
529 vecteurs->m[1].tail.y = working_vecteur.tail.y + dist_tail * sin(angle2);
530 vecteurs->m[1].tail.x =
531 working_vecteur.tail.x + l * dist_tail * cos(angle2);
532}
533double getMiddleAngle(DVecteur vecteurs) {
534 double middle_head =
535 getPointDistance(vecteurs.m[0].head, vecteurs.m[1].head) / 2.0;
536 double middle_tail =
537 getPointDistance(vecteurs.m[0].tail, vecteurs.m[1].tail) / 2.0;
538 double beta = atan2(vecteurs.m[1].head.y - vecteurs.m[0].head.y,
539 vecteurs.m[1].head.x - vecteurs.m[0].head.x);
540 Vecteur middle;
541 middle.head.x = vecteurs.m[0].head.x + cos(fabs(beta)) * middle_head;
542 middle.head.y = vecteurs.m[0].head.y + sin(beta) * middle_head;
543 double beta2 = atan2(vecteurs.m[1].tail.y - vecteurs.m[0].tail.y,
544 vecteurs.m[1].tail.x - vecteurs.m[0].tail.x);
545 middle.tail.x = vecteurs.m[0].tail.x + cos(fabs(beta2)) * middle_tail;
546 middle.tail.y = vecteurs.m[0].tail.y + sin(beta2) * middle_tail;
547 return atan2(middle.head.x - middle.tail.x, middle.head.y - middle.tail.y);
548}
